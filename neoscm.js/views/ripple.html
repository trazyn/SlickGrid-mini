Ripple Anything
API
HTML 
		
		<button class="ui button ripple">Ripple Example</button>
.ui.ripple 必须
Javascript 
		
		$( selector ).ripple( options );
默认绑定点击事件，用户点击触发动画 
		
		
		Default options:

		{
			/** 大于0时为持续性行为，间隔为该值，直到调用hide()结束，单位毫秒 */
			speed 		: 0,

			/** 看上面例子 */
			random 		: false,

			/** 调用hide()结束状态默认为success，该值为状态变更时所对应样式的持续时间 */
			delay 		: 300,

			/** show() 到 hide() 之间所对应的附加样式 */
			class4loading 	: "ripple-loading",
			/** hide( true ) 时候对应的附加样式 */
			class4lsuccess 	: "ripple-success",
			/** hide( false ) 时候对应的附加样式 */
			class4error 		: "ripple-error"
		}

		Methods:

		show() 				- 开始动画
		hide( [success] ) 		- 作用于附加样式，默认为true
