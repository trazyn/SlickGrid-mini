		$.message.success( content, [delay] );

Warn

Confirm
		
		$.message.confirm( options );

		Options:
		{
			title 		: 标题
			message 	: 消息内容
			onOk 		: 点击Ok时执行的回调,返回false时候不会关闭对话框
			onClose 		: 点击Cancle时候执行的回调
		}

		eg:

		$.message.confirm( {
			
			title: "Please confirm",
			message: "Exported successfully. Do you want to open the export query page?",
			onOk: function() {
				window.open( "//www.google.com", "_blank" );
			}
		} );
